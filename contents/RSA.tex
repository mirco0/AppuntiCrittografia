\begin{theorem}[Piccolo teorema di Fermat]
\label{thm:fermat}
Se $p$ è primo, allora, per ogni $a$, 
\[
    a^p \equiv a \pmod p.
    \]
    Equivalentemente, se $p$ è primo e $a$ è coprimo con $p$, ovvero $\gcd (a,p)=1$, allora 
    \[a^{p-1} \equiv 1 \pmod p.
    \] 
    
\end{theorem}
% (Tale teorema è alla base del test di primalità.) 
\begin{proof} (Dimostrazione come corollario del teorema di Lagrange sui gruppi)\\
    Essendo $p$ primo, gli interi da $1$ a $p-1$, quindi escluso lo zero, formano un gruppo moltiplicativo $G \pmod p$, ovvero $(\mathbb{Z}_p^*, \cdot)$, avente come elemento identità il numero $1$. \\ L'ordine del gruppo (il numero di elementi) è pari a $n-1$.
    
    \vspace{1em} L'ordine di un elemento qualsiasi nel gruppo, $a \in G$, è il più piccolo esponente $i$ tale che $a^i$ sia l'identità. \\ Ma, dal teorema di Lagrange sui gruppi, l'ordine dell'elemento deve dividere l'ordine del gruppo. \\ Ciò significa che $a^{p-1} \equiv 1 \pmod p$ e che l'esponente deve essere un multiplo di $p-1$. Quindi, elevando qualsiasi elemento del gruppo a $p-1$, si ottiene sempre $1 \pmod p$. 
\end{proof}

\paragraph{Generalizzazioni}
\begin{itemize}
    \item se $p$ è primo e $n,m >0$ con $m \equiv n \pmod {p-1}$, allora, $\forall a$, $a^m \equiv a^n \pmod p$.
    \item \textbf{Teorema di Eulero} Sia $n$ un intero, e $a$, coprimo con $n$: allora vale che $a^{\phi(n)} \equiv 1 \pmod n$.
\end{itemize}

\begin{corollary}
    Se $a \in {\mathbb{Z}_n}^*$, allora $a^{\phi(n)} \equiv 1 \pmod n$. \label{Corollario1}
\end{corollary}

\begin{proof}
    ${\mathbb{Z}_n}^*$ è un gruppo moltiplicativo di ordine $n$.
\end{proof}

\paragraph{Corollario (Fermat)} Sia $p$ primo e $a \in \mathbb{Z}_p$. Allora $a^p \equiv a \pmod p$. 

\begin{proof}
    Se $p$ è primo, allora $\phi(p)=p-1$. Quindi:
    \begin{itemize}
        \item per $b \not \equiv 0 \pmod p$, il risultato deriva da \ref{Corollario1}
        \item per $b \equiv 0 \pmod p$, poiché $0^p \equiv 0 \pmod p$, è vero.
    \end{itemize}
\end{proof}

\subsection{Introduzione alla Crittografia Asimmetrica}
Per evitare che Alice e Bob si mettano d'accordo in anticipo su una chiave comune, con il rischio di comunicare su un canale insicuro, nasce l'idea di crittografia a chiave pubblica. Alice usa una chiave pubblica $e_K$, disponibile a chiunque per la cifratura, mentre Bob decifra con la sua chiave privata $d_K$, posseduta solo da lui.

\vspace{0.2cm} \noindent Un'osservazione è che tale tipo di crittografia non può mai fornire una sicurezza incondizionata, perché un avversario, osservando il \emph{ciphertext} $y$, può cifrarlo usando $e_k$ finché non trova l'unica $x$ tale che $y=e_K(x)$.\\ Quindi decifrare dovrebbe essere difficile, tranne che per Bob.

\vspace{0.2cm} \noindent Potremmo prendere funzioni facili da calcolare, ma difficili da invertire come funzioni ``one-way'' iniettive, ma non è sufficiente, perché Bob dovrebbe decifrare in modo efficiente. È necessario quindi che Bob possieda una ``trapdoor'', una conoscenza segreta aggiuntiva, ovvero $K$, che gli permetta di invertire facilmente $e_K$. \\ È utile specificare una famiglia di funzioni ``one-way trapdoor'' $\mathcal{F}$; una funzione $f \in \mathcal{F}$ viene scelta casualmente ed usata per la cifratura pubblica; l'inversa $f^{-1}$ è la privata.

\subsection{Algoritmo Euclideo, Identità di Bézout e Teorema cinese del resto}
\subsubsection{Algoritmo Euclideo}
Si presenta ora l'algoritmo euclideo utilizzato per calcolare il valore $\gcd(a,b)$ per due dati interi, $a,b$. Si inizia impostando $r_0 = a$ e $r_1 = b$:
\[
    \begin{aligned}
        r_0 &= q_1 r_1 + r_2, \qquad 0 < r_2 < r_1\\
        r_1 &= q_2 r_2 + r_3, \qquad 0 < r_3 < r_2\\
        \vdots &\qquad \vdots \\
        r_{m-2} &= q_{m-1} r_{m-1} + r_m, \qquad 0 < r_m < r_{m-1}\\
        r_{m-1} &= q_m r_m
    \end{aligned}
\]
di seguito l'algoritmo iterato divide $r_{i-1}$ per $r_{i}$ per ogni $i=1,\dots,m$, è chiaro che:
\[
\gcd(r_0,r_1) = \gcd(r_1,r_2) = \ldots = \gcd(r_{m-1},r_m) = r_m
\] 

\input{algorithms/euclidean.tex}

poiché per ogni $b \in \mathbb{Z}_n$ esiste l'elemento inverso $b^{-1}$ esiste se $b$ è coprimo con $n$, ovvero $\gcd(b,n) = 1$, valore che l'algoritmo euclideo permette di calcolare.

\begin{theorem}
\label{thm:2}
Per $ 0 \leq j \leq m$ vale $r_j = s_j r_0 + t_j r_1$ con $(t_0,t_1,\dots,t_m)$ e $(s_0,s_1,\dots,s_m)$ due sequenze di numeri in cui: 
\[
    \begin{aligned}
    t_j &= \begin{cases}
    0 & \text{se } j = 0\\
    1 & \text{se } j = 1\\
    t_{j-2}-q_{j-1}t_{j-1} & \text{se } j \geq 2 
    \end{cases}
    \qquad
    s_j &= \begin{cases}
    1 & \text{se } j = 0\\
    0 & \text{se } j = 1\\
    s_{j-2}-q_{j-1}s_{j-1} & \text{se } j \geq 2 
    \end{cases}
    \end{aligned}
\]
e le $r_j$ sono definite come nell'algoritmo euclideo.
\end{theorem}

\vspace{1em}
\noindent Si procede a dimostrare per induzione
\begin{proof}
    Per $j = 0$ è chiaro che $r_0 = s_0 r_0 + t_0 r_1 = r_0  = 1 \cdot r_0 + 0 \cdot r_1 = \boxed{r_0}$, lo stesso vale per $j = 1$, $r_1 = 0 \cdot r_0 + 1 \cdot r_1 = \boxed{r_1}$.
    
    \vspace{1em}\noindent
    Si assume vero per $j = i-1$ e $j = i-2$ con $i-2 \geq 2$, si dimostra vero per $j = i$
    \begin{align*}        
        r_{i-2} &= s_{i-2}r_0 + t_{i-2}r_1\\
        r_{i-1} &= s_{i-1}r_0 + t_{i-1}r_1\\
        r_i &= r_{i-2} - q_{i-1}r_{i-1}\\
        &= s_{i-2}r_0 + t_{i-2}r_1 - q_{i-1}(s_{i-1}r_0 + t_{i-1}r_1)\\
        &= r_0(s_{i-2} - q_{i-1}s_{i-1}) + r_1(t_{i-2-q_{i-1}t_{i-1}})\\
        & = s_i r_0 + t_i r_1
    \end{align*}
    è perciò vero per $j \geq 0$.
\end{proof}
\begin{corollary}
    Siano $r_1, r_2$ interi tali che $\gcd(r_0,r_1) = 1$ allora $r_1^{-1} \equiv t_m \pmod{r_0}$ con $t_m$ dalla sequenza del teorema~\ref{thm:2}
\end{corollary}
\begin{proof}
    \begin{align*}
        \gcd(r_0,r_1) &= s_m r_0 + t_m r_1\\
        &= t_m r_1 \pmod n\\
        &= 1.
    \end{align*}
\end{proof}

\begin{theorem}[Teorema Cinese del resto]
    Siano $n_1,\dots,n_r$ numeri primi tra loro, allora il sistema
    \begin{equation}
        \begin{cases}
            x \equiv a_1 \pmod n_1\\
            x \equiv a_2 \pmod n_2\\
            \ \vdots \qquad \vdots\\
            x \equiv a_r \pmod n_r
        \end{cases}
    \end{equation}
ha soluzione unica modulo $N = n_1 \cdot n_r$, data dalla formula 
\[
    x = \sum_{i=1}^{r}a_i \frac{N_i} y_i \pmod N
\]
con $N_i = \frac{N}{n_i} e y_i = N_i^{-1}$.
\end{theorem}
\begin{proof}
    Si definisce una funzione $\chi: \mathbb{Z}_n \to \mathbb{Z}_{n_1} \times \cdots \times \cdots \mathbb{Z}_{r}$ chiamata funzione di proiezione, che, per una data sequenza $n_1,\dots,n_r$, e un valore $x$ restituisce una tupla con $r$ elementi:
    \[
        \chi(x) = (x \pmod n_1, \dots, x \pmod n_r),
    \]
    
    si vuole dimostrare che la funzione $\chi$ è una biezione. Per $1 \leq i \leq r$ 
    \begin{equation*}
        N_i := \frac{N}{n_i} \implies \gcd(N_i,n_i) = 1\\
    \end{equation*}
    per cui $N_i$ ha un inversa moltiplicativa modulo $n_i$, ovvero $y_i = {N_i}^{-1} \pmod n_i$. Si definisce ora una funzione $\rho:\mathbb{Z}_{n_1} \times \cdots \times \cdots \mathbb{Z}_{r} \to \mathbb{Z}_n$, che al contrario di $\chi$ da una tupla di $r$ elementi modulo $n_i$ restituisce un elemento in $\mathbb{Z}_N$:
    \[
        \rho(a_1,\dots,a_r) = \sum_{i=1}^{r}a_i N_i y_i \pmod N
    \]
    si considera il termine $m_j$ nella somma precedente, se $j = i$ allora poiché $y_i$ è l'inversa di $N_i$ modulo $n_i$ allora $N_i y_i \equiv 1 \pmod {n_i}$, per cui:  
    \[
        a_i N_i y_i \equiv a_i \pmod (n_i),
    \] al contrario se $j \neq i$ allora, poiché $n_j \mid N_i$
    \[
        a_i N_i y_i \equiv 0 \pmod {n_j}
    \]
    per cui nella funzione $\rho$, i termini con $n_j$ restano, mentre tutti gli altri si annullano
    \begin{align*}        
        \rho(a_1,\dots,a_r) &\equiv \sum_{i=1}^{r} a_i N_i y_i \pmod n_j\\
        &\equiv a_j \pmod{n_j}. 
    \end{align*}
    Poiché è sempre vero per $0\leq j \leq r$, $\rho$ è una soluzione al sistema di congruenze, modulo $N$. $\chi$ è quindi una funzione iniettiva, poiché ha un dominio di cardinalità $N$ in un codominio di cardinalità $N$. $\chi$ è una biezione e vale che $\chi^{-1} = \rho$.
    %da capire meglio
\end{proof}

\subsection{Crittosistema RSA} Sia $n=pq$, con $p,q$ primi. Sia $\mathcal{M}=\mathcal{C}=\mathbb{Z}_n$ e definiamo $\mathcal{K}=\{(n,p,q,a,b) : ab \equiv 1 \pmod \phi(n)\}$. \\ Per $\mathcal{K}=(n,p,q,a,b)$, definiamo: 
\[
e_K(x)=x^b \pmod n \ \text{e} \ d_K(y)=y^a \pmod n \ \text{con} \ (x,y \in \mathbb{Z}_n). 
\]
I valori $n$ e $b$ costituiscono la chiave pubblica, $p,q, a$ formano la chiave privata. 


\vspace{0.2cm} \noindent Verifichiamo ora che cifratura e decifratura sono operazioni inverse. Ovviamente $ ab \equiv 1 \pmod \phi(p)$ sta a significare che $a$ e $b$ sono una l'inversa dell'altra modulo $\phi(n)$, inoltre si ha che $ab =  {t \phi(n)} +1$ per qualche $t \geq 1$. \\ Supponiamo che $x \in {\mathbb{Z}_n}^*$, allora abbiamo: 

\begin{align*}
    (x^b)^a &\equiv x^{t\phi(n)+1} \pmod n \\
    &\equiv (x^{\phi(n)t})x \pmod n \\
    &\equiv (1^t)x \pmod n \\
    &\equiv x \pmod n 
\end{align*} come richiesto. 



\paragraph{Generazione dei parametri RSA} 
    \begin{enumerate}
        \item Siano $p,q$ due primi diversi, sufficientemente ``grandi''
        \item $n \gets pq$ e $\phi(n) \gets (p-1)(q-1)$
        \item Scegli $b$ a caso $1<b<\phi(n)$ tale che $\gcd(b,\phi(n))=1$
        \item $a \gets b^{-1} \pmod {\phi(n)}$
        \item La chiave pubblica è $(n,b)$ e la chiave privata è $(p,q,a)$.  
    \end{enumerate}

    Affinché il crittosistema RSA sia sicuro, è necessario scegliere $n=pq$ tanto grande, da rendere computazionalmente impossibile la sua fattorizzazione. Attualmente, si consiglia di scegliere $p,q$ a $1024$ bit, cosicché $n$ ne abbia $2048$. Trovare numeri primi grandi non è un compito banale, per ora supponiamo di averne già alcuni a disposizione.
    
    \vspace{1em}\noindent 
    Supponiamo che $x$ e $y$ siano due interi con, rispettivamente, $k$ ed $\ell$ bit, con $k \geq \ell$:
    \begin{align*}
        x = \underbrace{{10\dots00}}_{k = \lfloor \log_2{x} \rfloor + 1} \quad y = \underbrace{{11\dots10}}_{\ell = \lfloor \log_2{y} \rfloor + 1}\\
    \end{align*}
    Da algoritmi banali è facile determinare le complessità delle operazioni, nella aritmetica modulare si ottiene un'analisi molto simile con le operazioni in $\mathbb{Z}_n $ e due elementi $0 \leq m_1,m_2 \leq n-1$, in cui $n$ è un intero a $k$ bit:

            \textit{Operazione} & \textit{Costo}\\
            \hline
            $x + y$ & $O(k)$\\
            $x-y$ & $O(k)$\\

\subsection{Test di primalità}
Si tratta ora il problema di determinare numeri primi sufficientemente grandi, che permettono al sistema RSA di essere sicuro

\begin{definition}{Problema Composti}
    Dato un numero $n$, determinare se è primo
\end{definition}

Ci sono diversi approcci per risolvere il problema, gli algoritmi proposti spesso sono basati sulla casualità, si dividono in due categorie: algoritmi \emph{Las Vegas}, di cui abbiamo già parlato e algoritmi del tipo \emph{Monte Carlo}

\begin{definition}
    Un algoritmo di Monte Carlo\footnote{Un algoritmo di tipo Monte Carlo dà sempre una risposta, che però potrebbe non essere corretta. \\ Diversamente da un algoritmo di tipo Las Vegas, che potrebbe non rispondere, ma se dà una risposta è sicuramente corretta.} \emph{yes-biased} è un algoritmo il cui risultato è sempre corretto nel caso di risposta ``si'', mentre un risultato del tipo ``no'' può essere errato; nel caso contrario si parla algoritmo di Monte Carlo \emph{no-biased}. Un algoritmo di Monte Carlo, ha probabilità di errore $\varepsilon$ se per ogni istanza, il risultato è errato con una probabilità di al massimo $\varepsilon$.   
\end{definition}

\begin{definition}[Residui quadratici]
    Sia $p>2$ un primo, e $a$ un intero. $a$ è chiamato \emph{residuo quadratico} modulo $p$ se 
    \[
        a \not\equiv 0 \pmod p \quad \land \quad  y^2 \equiv a \pmod p 
    \]
    per una $y \in \mathbb{Z}_p$; nel caso in cui non esista $y$, $a$ è detto \textit{non residuo quadratico}.
\end{definition}
Ovviamente se $a$ è un residuo quadratico modulo $p$ allora esiste $y \in {\mathbb{Z}_p}^*$ per cui $y^2 \equiv a$ e
\begin{align*}
    y^2 \equiv a \implies (-y^2) \equiv a \pmod p\\
    y \not\equiv -y \pmod p
\end{align*}
inoltre
\begin{gather*}
    x^2 - a \equiv 0 \\
    x^2 - y^2 \equiv 0 \\
    (x-y)(x+y) \equiv 0 \\
    p|(x-y) \quad \text{oppure} \quad p |(x+1)\\
    x \equiv \pm y 
\end{gather*}
\noindent Quindi, si conclude che $x^2 -a \equiv 0 \pmod p$ ha esattamente $2$ soluzioni modulo $p$. 

% \newpage
\vspace{0.2cm}\noindent Dato $p$ primo e $a$ un intero, come capire se $a$ è residuo quadratico modulo $p$?

\begin{theorem}[Criterio di Eulero]
    \label{thm:3}
    Sia $p$ un numero primo, allora $a$ è un residuo quadratico \textbf{se e solo se} 
    \[
        a^{(p-1)/2} \equiv 1 \pmod p
    \]
\end{theorem}
\begin{proof}
    ($\Leftarrow$) Supponiamo che $a$ sia un residuo quadratico, per cui vale che $a \equiv y^2$ per qualche $y \in {\mathbb{Z}_p}^*$, banalmente si ha che 
    \begin{align*}
        a^{(p-1)/2} &\equiv (y^{2})^{(p-1)/2} \pmod p\\
        &\equiv y^{p-1} \pmod p\\
        \substack{\text{Per il teorema} \\\ref{thm:fermat}}  & \equiv 1 \pmod p.
    \end{align*}
    Ricordando che il teorema di Fermat \ref{thm:fermat}.

    \vspace{1em} ($\Rightarrow$)
    Al contrario sia valido $a^{(p-1)/2} \equiv 1 \pmod p$, allora sia $b$ un elemento primitivo modulo $p$, per cui è possibile scrivere qualsiasi elemento in ${\mathbb{Z}_p}^*$ come una potenza $i$-esima di $b$ ($a = b^i$).
    \begin{align*}
        a^{(p-1)/2} &\equiv (b^i)^{(p-1)/2} \pmod p\\
        &\equiv b^{i(p-1)/2} \pmod p 
    \end{align*}
    %TODO: Capire meglio
    ovviamente $b$, essendo primitivo, ha ordine $p-1$, quindi $p-1$ divide $i(p-1)/2$, e $i$ perciò è pari; le radici di $a$ sono $\pm b^{i/2} \pmod p$.
%mi sa ho sbagliato i versi delle frecce bah :)
\end{proof}

\begin{theorem}
    Sia $p>2$ un numero primo e $a \in \mathbb{Z}_p$ un intero. Il numero di Legendre $\left(\frac{a}{p}\right)$ è definito come segue:
    \[
        \left( \frac{a}{p}\right) = \begin{cases}
            0 & \text{se } a \equiv 0 \pmod p\\
            1 & \text{se $a$ è un residuo quadratico modulo $p$}\\
            -1 & \text{se $a$ è un non-residuo quadratico modulo $p$}.
        \end{cases}
    \] Allora vale la seguente proprietà:
    \[
        \left(\frac{a}{p}\right) \equiv a^{(p-1)/2} \pmod p .
    \]
\end{theorem}

\begin{proof}
    Partendo dal primo caso, assumiamo $a$ sia un multiplo di $p$. Chiaramente  $a \equiv 0 \pmod p$, e ovviamente 
    \[
        a^{(p-1)/2} \equiv 0^{(p-1)/2} \equiv 0 \pmod p .
    \]
    Supponendo invece che $a$ sia un residuo quadratico, dal teorema \ref{thm:3} è immediato 
    \[
        a^{(p-1)/2} \equiv 1 \pmod p .
    \]
    Come ultimo caso analizziamo il caso in cui $a$ sia un non-residuo quadratico, per cui $a^{(p-1)/2} \not\equiv 1 \pmod p$ e 
    \begin{gather}        
        (a^{(p-1)/2})^2 \equiv a^{p-1} \equiv 1 \pmod p \quad \substack{\text{per il teorema} \\ \ref{thm:fermat}}\\  
        a^{(p-1)/2} \not\equiv 1 \implies a^{(p-1)/2} \equiv -1 \pmod p . \label{eq4}
    \end{gather}
    in \ref*{eq4} poiché il quadrato è $1$, ma la radice è diversa da $1$ allora, il valore cercato è proprio $-1$. In tutti e tre i casi i risultati sono quindi verificati.
\end{proof}

\noindent Un altro risultato importante riguardante il numero di \emph{Legendre} è la seguente generalizzazione:
\begin{definition}[Simbolo di Jacobi]
    Per un intero $n$, di cui la fattorizzazione è
    \[
        n = \prod_{i=1}^{k}p_i^{e_i}.
    \]
    e, per un intero $a$, il simbolo di Jacobi è definito come:
    \[
        \left( \frac{a}{n}\right) = \prod_{i=1}^{k} \left(\frac{a}{p_i}\right)^{e_i}.
    \]
    Il simbolo di Jacobi per un $n$ composto è quindi calcolabile come prodotto di simboli di Legendre dei primi che lo compongono.
\end{definition}

\noindent
Dato il simbolo di Jacobi è ora possibile determinare la primalità di un numero con la seguente proprietà:
\begin{theorem}
    Per $p$ primo, il numero di quadrati residui nel gruppo ${\mathbb{Z}_p}^*$ è esattamente $\frac{p-1}{2}$.
\end{theorem}
\begin{proof}
    Per il gruppo ${\mathbb{Z}_p}^*$ esiste un generatore $g$ tale che: 
    \[ 
        {\mathbb{Z}_p}^* = \langle g,g^2,\dots,g^{\phi(p)} \rangle
    \]
    ${\mathbb{Z}_p}^*$ è ciclico di ordine $\phi(p)$. Per $p$ primo $\phi(p) = p-1$, i residui quadratici sono tutti gli elementi del gruppo, generati con un esponente \emph{pari} ($g^{2k}$); per cui i residui quadrati sono esattamente $\frac{p-1}{2}$.

\end{proof}

Sia $b$ una base scelta casualmente in $[2, n-2]$. Considerando la sequenza delle potenze:
\[ b^m, b^{2m}, \dots, b^{2^km} \]
Se $n$ è primo, per il teorema di Fermat \ref{thm:fermat} $b^{n-1} \equiv 1 \pmod n$.
\begin{align*}
    b^{n-1} &\equiv 1 \pmod n = b^{2^km} \equiv 1 \pmod n\\
    b^{(n-1)/2} &\equiv \pm 1 \pmod n = b^{2^{k-1}} \equiv \pm 1 \pmod n
\end{align*}

L'idea dell'algoritmo è quella di verificare gli elementi del gruppo, dati da una base $b$, con una potenza pari ($b^2,b^4,\dots$), in questo caso è richiesto che se $b^{2^km}\equiv 1$ allora la sua radice quadrata $b^{2^{k-1}m}$ deve essere $\pm 1$, il test supera la verifica (ovvero $n$ è un \emph{probabile primo}) se:
\begin{itemize}
    \item $b^m \equiv 1 \pmod n$, oppure
    \item Esiste un indice $i \in \{0, \dots, k-1\}$ tale che $b^{2^i m} \equiv -1 \pmod n$.
\end{itemize}
Se nessuna di queste condizioni è verificata, $n$ è sicuramente composto. Per cui l'algoritmo di Miller Rabin restituisce la soluzione corretta in caso $n$ sia composto, la probabilità che l'algoritmo sbagli per un numero $n$ primo è $\leq \frac{1}{4}$.

\begin{theorem}
    Sia $p>2$ primo, $e>0$, e $a$ un intero tale che $\gcd(a,n) = 1$. Allora la  congruenza $y^2 = a \pmod {p^e}$ ha:~
    \begin{itemize}
        \item 0 soluzioni se $\left( \frac{a}{p}\right) = -1$
        \item 2 soluzioni se $\left( \frac{a}{p}\right) = 1 \pmod {p^e}$
    \end{itemize}
\end{theorem}

\begin{theorem}
    Sia $n>1$, $e>0$, e $a$ un intero tale che $\gcd(a,n) = 1$. La fattorizzazione di $n$ è:
    \[
        n = \prod_{i=1}^{\ell} p_i^{e_i},
    \]
    in cui $p_i$ sono i fattori primi di $n$ ed $e_i$ le rispettive potenze. La congruenza $y^2 = a \pmod {p}$ ha:
    \begin{itemize}
        \item $2^{\ell}$ soluzioni $\pmod n$ se $\left( \frac{a}{p_i}\right) = 1 \quad \forall i=0,\dots,\ell$
        \item 0 soluzioni altrimenti
    \end{itemize}

\end{theorem}
\begin{proof}
    È ovvio che $y^2 \equiv a \pmod n$ ha soluzione per ogni $y^2 \equiv a \pmod {p_i^{e_i}}$. Supponendo che $\left(\frac{a}{p_i}\right) = 1$ per ogni $i = 1,\dots, \ell$, allora ognuna delle equazioni ha due soluzioni $b_{i1}, b_{i2}$; si definisce $b_i=\{b_{i1},b_{i2}\}$. Il sistema di congruenze:
    \begin{align*}
        {y_1} &\equiv b_1 \pmod {p_1^{e_1}}\\
        \vdots &\quad \vdots \quad \vdots\\
        {y_\ell} &\equiv b_\ell \pmod {p_\ell^{e_\ell}}
    \end{align*}
    ha un unica soluzione per il teorema cinese del resto. I sistemi possibili sono $2^{\ell}$, per cui ci sono $2^\ell$ possibili soluzioni uniche modulo $n$ per la congruenza $y^2 \equiv a \pmod n$.
    
\end{proof}

    &\equiv a(y^{-1})^2 \pmod n \\  
    &\equiv a \cdot a^{-1} \pmod n \\

\end{aligned}

(x z)^2 \equiv x^2 z^2 \equiv x^2 \pmod n.
\]
Questo significa che, moltiplicando una radice quadrata di $a$ per una radice quadrata di $1$, otteniamo un’altra radice quadrata dello stesso elemento $a$. \\ Di conseguenza, se ci sono $2^\ell$ radici quadrate di $1$ modulo $n$, allora a partire da una radice quadrata fissata di $a$ possiamo generare tutte le $2^\ell$ radici quadrate di $a$.

\noindent Nell'input, $B$ rappresenta un ``limite'' prestabilito, ovvero ``quanto grandi'' possono essere i fattori primi di $(p-1)$.

\vspace{0.2cm} \noindent Idea base: Supponiamo che $p$ sia un fattore primo di $n$.
%q? q<B pagina 231 pdf 
Se tutti i fattori primi di $p-1$ sono minori o uguali a $B$, ovvero ``piccoli'', allora $(p-1) | B!$ . \\ Alla fine del \textit{for} si ha: 
\[
\begin{aligned}
    &a \equiv 2^{B!} \pmod n\\
    &a \equiv 2^{B!} \pmod p \ \text{(poiché $p|n$)} \\
    &2^{p-1} \equiv 1 \pmod p \ \text{(per il Teorema di Fermat)} \\
    & a \equiv 1 \pmod p \ \text{(poiché $(p-1) | B!$)}.\\
\end{aligned}
\]

\noindent In altre parole $p | (a-1)$. Poiché $p|n$, otteniamo:
\[
d= \gcd(a-1,n).
\]
Se $a \neq 1$, $d$ è un fattore non banale di $n$. Se $d$ e $n/d$ sono ancora composti, si itera l'algoritmo per ottenere la fattorizzazione completa. 
 
\vspace{0.2cm}\noindent \textbf{Complessità:}  
\begin{itemize}
    \item Ci sono $(B-1)$ esponenziazioni modulari; ciascuna richiede $O(\log B)$ operazioni con \textit{square-and-multiply}.
    \item Il calcolo del $\gcd(a-1,n)$ richiede $O((\log n)^3)$ usando l’algoritmo euclideo.
    \item Dunque la complessità totale è
    \[
    O\big(B \log B (\log n)^2 + (\log n)^3\big)
    \]
\end{itemize}
\noindent Se $B\approx \sqrt{n}$, ovvero molto grande, l'algoritmo garantisce successo, ma diventa lento quanto la divisione per tentativi. \\ Se $(p-1)$ contiene fattori primi grandi, l'algoritmo rischia di fallire. 

\vspace{0.2cm}\noindent Dunque, un punto debole di tale algoritmo è che richiede che $n$ abbia un fattore primo $p$ tale che $(p-1)$ abbia fattori primi ``piccoli''.
