\begin{algorithm}[H]
    \SetAlgoVlined
    \SetAlgoLined
    
    \caption{Trova Seconda Preimmagine($h,x,Q$)}
    \label{alg:second_preimage}
    
    \KwIn{Parametri $h, x, Q$}
    \KwOut{Elemento $y$ o errore}


    $y \gets h(x)$\;
    scegli $\mathcal{X}_0 \subseteq \mathcal{X}-\{x\}$ tale che $|\mathcal{X}_0| = Q - 1$\;
    \ForEach{$x_0 \in \mathcal{X}_0$}{
        \If{$h(x_0) = y$}{
            \Return{$x_0$}\;
        }
    }
    \Return{errore}\;
\end{algorithm}
