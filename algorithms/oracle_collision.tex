\begin{algorithm}[H]

    \SetAlgoVlined
    \SetAlgoLined
    \caption{Trova Collisione($h,Q$)}
    \label{alg:collision}

    \KwIn{Parametri $h, Q$}
    \KwOut{Coppia $(x,x')$ o errore}
    
    scegli $\mathcal{X}_0 \subseteq \mathcal{X}$ tale che $|\mathcal{X}| = Q$\;
    \ForEach{$x \in \mathcal{X}_0$}{
        $y_x \gets h(x)$\;
    }
    \If{$y_x = y_{x'}$ per qualche $x' \neq x$}{
        \Return{$(x,x')$}\;
    }
    \Return{errore}\;
\end{algorithm}
