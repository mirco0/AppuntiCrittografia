    \begin{algorithm}[H]
        \label{alg:collision}
        \DontPrintSemicolon
        \TitleOfAlgo{Trova Preimmagine($h,y,Q$)}
        scegli $\mathcal{X}_0 \subseteq \mathcal{X}$ tale che $|\mathcal{X}| = Q$\;
        \ForEach{$x \in \mathcal{X}_0$}{
            Calcola $y_x \gets h(x)$\;
        }
        \If{$y_x = y_{x'}$ per qualche $x' \neq x$}{
            \Return{$x$}\;
        }
        \Return{Fail}\;
    \end{algorithm}
