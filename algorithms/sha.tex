\begin{algorithm}[H]
    \SetAlgoVlined
    \SetAlgoLined
    
    \caption{Cifrario a blocchi SHA($t ,k$)}
    \label{alg:SHA256}
    
    \KwIn{Parametri $t,k$}
    \KwOut{Ciphertext}

    \ForEach{$i=0$ \KwTo  $15$}{
        $W_i \gets k_i$\;
    }

    \ForEach{$i=16$ \KwTo  $63$}{
        $W_i \gets \sigma_1(W_{i-2}) + W_{i-7} + \sigma_0(W_{i-15}) + W_{i-16}$\;
    }

    $(a_0,b_0,c_0,d_0,e_0,f_0,g_0,h_0) \gets (t_0,t_1,t_2,t_3,t_4,t_5,t_6,t_7)$\;
    \ForEach{$i=0$ \KwTo  $63$}{
        $T_1 \gets h_i + \Sigma_i(e_i) + \text{Ch}(e_i,f_i,g_i) + K_i + W_i$\;
        $T_2 \gets \Sigma_0(a_i) + \text{Maj}(a_i,b_i,c_i)$\;
        $(a_{i+1},b_{i+1},c_{i+1},d_{i+1},e_{i+1},f_{i+1},g_{i+1},h_{i+1}) \gets (T_1 + T_2, a_i, b_i,c_i, d_i + T_i, e_i, f_i,g_i)$

    }

    \Return{$a_{64}  \conc  b_{64}  \conc  c_{64}  \conc  d_{64}  \conc  e_{64}  \conc  f_{64}  \conc  g_{64}  \conc  h_{64}$}\;
\end{algorithm}
