\begin{algorithm}[H]
    \SetAlgoVlined
    \SetAlgoLined
    \caption{Inversa Moltiplicativa($a,b$)}
    \label{alg:euclideo_esteso}

    $a_0 \gets a, \quad b_0 \gets b, \quad t_0 \gets 0, \quad t \gets 1, \quad q \gets \lfloor \frac{a_0}{b_0}\rfloor$\;
    $r \gets a_0 - qb_0$\;
    
    \While{$r > 0$}{
        $ \mathit{temp} \gets t_0 - qt  $\;
        $ t_0 \gets t$\;
        $ t \gets \mathit{temp}$\;
        $ a_0 \gets b_0$\;
        $ b_0 \gets r$\;
        $ q \gets \lfloor \frac{a_0}{b_0} \rfloor $\;
        $ r \gets a_0 - qb_0$\;
    }
    \uIf{$b_0 = 1$}{
        \Return{$t$}\;
    }
    \Else{
        \Return{\emph{Inversa inesistente}}\;
    }
\end{algorithm}
