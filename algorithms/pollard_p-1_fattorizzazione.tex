\begin{algorithm}[H]
    \SetAlgoVlined
    \SetAlgoLined
    
    \caption{Pollard $p-1$($n,B$)}
    \label{alg:Pollard_p-1}
  
    $a \gets 2$\;
    
    \For{$j \gets 2$ \KwTo $B$}{
        $a \gets a^j \bmod n$\;
    }

    $d \gets \gcd(a-1, n)$\;

    \uIf{$ 1<d<n$}{
        \Return{$d$}\;
    }
    \Else{
        \Return{``Errore''}\;
    }
\end{algorithm}
