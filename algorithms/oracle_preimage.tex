\begin{algorithm}[H]
    \SetAlgoVlined
    \SetAlgoLined
    
    \caption{Trova Preimmagine($h,y,Q$)}
    \label{alg:preimage}
    
    \KwIn{Parametri $h, y, Q$}
    \KwOut{Elemento $x$ o errore}
    
    \BlankLine
    scegli $\mathcal{X}_0 \subseteq \mathcal{X}$ tale che $|\mathcal{X}_0| = Q$\;
    
    \ForEach{$x \in \mathcal{X}_0$}{
        \If{$h(x) = y$}{
            \Return{$x$}\;
        }
    }
    
    \Return{errore}\;
\end{algorithm}
