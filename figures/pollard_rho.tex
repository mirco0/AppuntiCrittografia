\begin{center}
\begin{tikzpicture}[
    >={Stealth[length=2.2mm]},
    thick,
    dot/.style={circle, fill=black, inner sep=0pt, minimum size=4.5pt},
    every node/.style={font=\small}
]

\def\R{1.5} % Raggio del cerchio
\coordinate (C) at (1.8,1.6); % Centro

%Coda
\node[dot,label=left:$x_1$] (x1) at (-2.1,-2.2) {};
\node[dot,label=left:$x_2$] (x2) at (-1.7,-1.3) {};
\node[dot,label=left:$\dots$] (xt) at (-0.7,0.3) {};
\node[dot,label=left:$x_i$] (xi) at ($(C)+(-180:\R)$) {};

\draw[->] (x1) to[out=60,in=240] (x2);
\draw[->] (x2) to[out=60,in=240] (xt);
\draw[->] (xt) to[out=60,in=230] (xi);

\foreach \angle/\name/\pos in {
    120/{x_{i+1}}/above left,
     50/{x_{i+2}}/above right,
    -10/{\dots}/right,
    -90/{x_{j-1}}/below,
   -180/{x_j=x_i}/right
}{
    \node[dot,label=\pos:$\name$] (p\angle) at ($(C)+(\angle:\R)$) {};
}

%Ciclo
\draw[->] (xi) arc (180:120:\R);
\draw[->] (p120) arc (120:50:\R);
\draw[->] (p50) arc (50:-10:\R);
\draw[->] (p-10) arc (-10:-90:\R);
\draw[->] (p-90) arc (-90:-180:\R);

\draw[->, thin, gray]
($(C)+(130:1.3)$) 
arc (130:-90:1.3)
node[midway, right=2mm, font=\footnotesize] 
{Ciclo di lunghezza $\ell$};

\end{tikzpicture}
